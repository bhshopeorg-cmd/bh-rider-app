// change this to your own secret, then update it in the HTML file too var SHEET_ORDERS = "Orders"; var SHEET_RIDERS = "Riders"; // ---------- HELPERS ---------- function getSheet_(name) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); if (name === SHEET_ORDERS) { sh.appendRow(["OrderID","CustomerName","Phone","Address","Amount","RiderEmail","Status","CashCollected","CreatedAt","DeliveredAt"]); } else if (name === SHEET_RIDERS) { sh.appendRow(["Email","Name","Phone","Status","RegisteredAt"]); } } return sh; } function sheetToObjects_(sheet) { var data = sheet.getDataRange().getValues(); var headers = data[0]; var rows = []; for (var i = 1; i < data.length; i++) { var obj = {}; for (var j = 0; j < headers.length; j++) obj[headers[j]] = data[i][j]; obj._row = i + 1; rows.push(obj); } return rows; } function jsonOut_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function findRowByValue_(sheet, colIndex, value) { var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][colIndex]).toLowerCase() === String(value).toLowerCase()) return i + 1; } return -1; } // ---------- GET ---------- function doGet(e) { var action = e.parameter.action; if (action === "bootstrap") { var email = (e.parameter.email || "").toLowerCase(); var riders = sheetToObjects_(getSheet_(SHEET_RIDERS)); var found = null; for (var i = 0; i < riders.length; i++) { if (String(riders[i].Email).toLowerCase() === email) { found = riders[i]; break; } } if (!found) return jsonOut_({ registered: false, approved: false }); return jsonOut_({ registered: true, approved: (found.Status === "Approved"), name: found.Name }); } if (action === "riderOrders") { var email2 = (e.parameter.email || "").toLowerCase(); var all = sheetToObjects_(getSheet_(SHEET_ORDERS)); var mine = all.filter(function(o) { return String(o.RiderEmail).toLowerCase() === email2; }) .map(function(o) { return { orderId: o.OrderID, customerName: o.CustomerName, phone: o.Phone, address: o.Address, amount: o.Amount, status: o.Status, cashCollected: o.CashCollected, createdAt: o.CreatedAt }; }); return jsonOut_({ orders: mine }); } if (action === "ownerData") { if (e.parameter.key !== OWNER_KEY) return jsonOut_({ error: "unauthorized" }); var orders = sheetToObjects_(getSheet_(SHEET_ORDERS)); var ridersList = sheetToObjects_(getSheet_(SHEET_RIDERS)); return jsonOut_({ orders: orders, riders: ridersList }); } if (action === "approvedRiders") { if (e.parameter.key !== OWNER_KEY) return jsonOut_({ error: "unauthorized" }); var r = sheetToObjects_(getSheet_(SHEET_RIDERS)).filter(function(x){ return x.Status === "Approved"; }); return jsonOut_({ riders: r }); } return jsonOut_({ error: "unknown action" }); } // ---------- POST ---------- function doPost(e) { var body = {}; try { body = JSON.parse(e.postData.contents); } catch (err) { return jsonOut_({ error: "bad request" }); } var action = body.action; if (action === "registerRider") { var email = String(body.email || "").toLowerCase().trim(); var name = body.name || ""; var phone = body.phone || ""; if (!email) return jsonOut_({ error: "email required" }); var sh = getSheet_(SHEET_RIDERS); var row = findRowByValue_(sh, 0, email); if (row > 0) { var status = sh.getRange(row, 4).getValue(); return jsonOut_({ ok: true, alreadyRegistered: true, approved: status === "Approved" }); } sh.appendRow([email, name, phone, "Pending", new Date()]); return jsonOut_({ ok: true, alreadyRegistered: false, approved: false }); } if (action === "approveRider") { if (body.key !== OWNER_KEY) return jsonOut_({ error: "unauthorized" }); var sh2 = getSheet_(SHEET_RIDERS); var row2 = findRowByValue_(sh2, 0, String(body.email || "").toLowerCase()); if (row2 < 0) return jsonOut_({ error: "rider not found" }); sh2.getRange(row2, 4).setValue("Approved"); return jsonOut_({ ok: true }); } if (action === "revokeRider") { if (body.key !== OWNER_KEY) return jsonOut_({ error: "unauthorized" }); var sh2b = getSheet_(SHEET_RIDERS); var row2b = findRowByValue_(sh2b, 0, String(body.email || "").toLowerCase()); if (row2b < 0) return jsonOut_({ error: "rider not found" }); sh2b.getRange(row2b, 4).setValue("Pending"); return jsonOut_({ ok: true }); } if (action === "addOrder") { if (body.key !== OWNER_KEY) return jsonOut_({ error: "unauthorized" }); var sh3 = getSheet_(SHEET_ORDERS); var orderId = "BH" + Utilities.formatDate(new Date(), Session.getScriptTimeZone() || "GMT", "yyMMddHHmmss"); sh3.appendRow([ orderId, body.customerName || "", body.phone || "", body.address || "", body.amount || 0, String(body.riderEmail || "").toLowerCase(), "Pending", false, new Date(), "" ]); return jsonOut_({ ok: true, orderId: orderId }); } if (action === "updateOrderStatus") { var email3 = String(body.email || "").toLowerCase(); var sh4 = getSheet_(SHEET_ORDERS); var row4 = findRowByValue_(sh4, 0, body.orderId); if (row4 < 0) return jsonOut_({ error: "order not found" }); var riderOnRow = String(sh4.getRange(row4, 6).getValue()).toLowerCase(); if (riderOnRow !== email3) return jsonOut_({ error: "unauthorized" }); sh4.getRange(row4, 7).setValue("Delivered"); sh4.getRange(row4, 8).setValue(body.cashCollected ? true : false); sh4.getRange(row4, 10).setValue(new Date()); return jsonOut_({ ok: true }); } return jsonOut_({ error: "unknown action" }); }
